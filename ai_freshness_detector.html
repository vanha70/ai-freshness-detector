Mã nguồn Python (Streamlit)
# Mở Terminal / Command Prompt và chạy lệnh này: pip install streamlit opencv-python numpy pillow tensorflow # Sau đó chạy app: streamlit run app.py # Trình duyệt tự mở tại: http://localhost:8501
import streamlit as st import cv2 import numpy as np from PIL import Image # ─── GIAO DIỆN ──────────────────────────────────────────────── st.set_page_config( page_title="AI Cảnh Báo Thực Phẩm", page_icon="🔬", layout="wide" ) st.title("🔬 Hệ Thống AI Nhận Diện Độ Tươi Thịt/Cá") st.caption("Phân tích không gian màu HSV từ màng sinh học Anthocyanin") col1, col2 = st.columns([1, 1], gap="large") with col1: uploaded_file = st.file_uploader( "📷 Tải ảnh chụp màng chỉ thị lên", type=["jpg", "jpeg", "png"] ) if uploaded_file is not None: # ─── ĐỌC VÀ HIỂN THỊ ẢNH ───────────────────────────────── image = Image.open(uploaded_file) with col1: st.image(image, caption="Ảnh màng chỉ thị", use_column_width=True) # ─── PHÂN TÍCH HSV ──────────────────────────────────────── img_array = np.array(image.convert("RGB")) hsv = cv2.cvtColor(img_array, cv2.COLOR_RGB2HSV) avg_h = np.mean(hsv[:, :, 0]) # Hue: 0-180 avg_s = np.mean(hsv[:, :, 1]) # Saturation: 0-255 avg_v = np.mean(hsv[:, :, 2]) # Value: 0-255 # ─── HIỂN THỊ KẾT QUẢ ──────────────────────────────────── with col2: st.subheader("📊 Thông số màu HSV") m1, m2, m3 = st.columns(3) m1.metric("Hue (H)", f"{avg_h:.1f}") m2.metric("Saturation (S)", f"{avg_s:.1f}") m3.metric("Value (V)", f"{avg_v:.1f}") # ─── LOGIC PHÂN LOẠI ────────────────────────────────── st.subheader("🤖 Kết quả AI") if 120 <= avg_h <= 180: st.success("✅ THỰC PHẨM CÒN TƯƠI\n\n" "Màng giữ màu tím/đỏ → môi trường axit\n" "TVB-N ước tính: < 15 mg/100g (An toàn)") elif 60 <= avg_h < 120: st.warning("⚠️ THỰC PHẨM CHỚM HỎNG\n\n" "Màng ngả xanh lục → NH₃ tăng (bazơ yếu)\n" "TVB-N ước tính: 15–25 mg/100g (Cảnh báo)") else: st.error("🚨 NGUY HIỂM: THỰC PHẨM ĐÃ HỎNG!\n\n" "Màng chuyển vàng → bazơ mạnh (NH₄OH)\n" "TVB-N ước tính: > 25 mg/100g (Không ăn!)") # Thanh tiến trình Hue st.progress(int(avg_h / 180 * 100), text=f"Hue = {avg_h:.1f} / 180") # ─── SIDEBAR THÔNG TIN ──────────────────────────────────────── st.sidebar.title("ℹ️ Về Dự Án") st.sidebar.info("Màng PVA-Anthocyanin kết hợp\n" "AI phân tích HSV để cảnh báo\n" "độ tươi hỏng thực phẩm.\n\n" "Đề tài KHKT cấp tỉnh/quốc gia.")
# Thêm đoạn này vào app.py để dùng mô hình Teachable Machine # (Export từ Teachable Machine → TensorFlow → Keras → file .h5) from tensorflow.keras.models import load_model # Load mô hình đã train (đặt file keras_model.h5 cùng thư mục) MODEL_PATH = "keras_model.h5" LABELS = ["🟢 Còn Tươi", "🟡 Chớm Hỏng", "🔴 Đã Hỏng"] @st.cache_resource def load_ai_model(): return load_model(MODEL_PATH, compile=False) model = load_ai_model() # Trong phần xử lý ảnh, thêm: def predict_freshness(pil_image): img = pil_image.resize((224, 224)) img_arr = np.array(img, dtype=np.float32) / 255.0 img_arr = np.expand_dims(img_arr, axis=0) preds = model.predict(img_arr)[0] best = np.argmax(preds) return LABELS[best], preds # Gọi trong app: label, confidences = predict_freshness(image) st.success(f"Kết quả: {label}") for i, conf in enumerate(confidences): st.progress(float(conf), text=f"{LABELS[i]}: {conf*100:.1f}%")
